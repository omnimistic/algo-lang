START
DECLARE count, secret
count = 1
secret = 0
OUTPUT "--- Starting the Loop ---"
OUTPUT count
IF count == 3 OUTPUT "We are exactly halfway there!"
IF count == 5 secret = 42
count = count + 1
IF count <= 5 GOTO 5
OUTPUT "--- Loop Finished ---"
OUTPUT "The secret number is:"
OUTPUT secret
IF secret != 0 OUTPUT "SUCCESS: Inline variable assignment worked!"
END
